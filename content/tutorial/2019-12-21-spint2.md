  * (Instantiation Patterns)

  * 3. Prototypal
  * 4. Pseudoclassical



* 1. Functional


```js

var Monster = function() {                
	var body = {};   
    body.level = 0;     
    body.levelUp = function() {    // Instance에 levelUp 라는 메소드를 추가
    	this.level = this.level + 1  // function내부에 있는 this는 instance를 뜻하므로 this.level 은 body.level과 같다.
    }                                 // 실행시 level 값이 1만큼 증가
    return body;
};




```
