  • 2개의 문자열이 주어졌을 때, areSameLength는 문자열들의 길이가 같으면 true를, 아니면 false를 반환해야 합니다.

    function areSameLength(string1, string2) {
	if (string1.length === string2.length) {
	  return true;
	} else {
return false;
}
}
