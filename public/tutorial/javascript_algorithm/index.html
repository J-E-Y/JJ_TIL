function nonUniqueElements(data) {
    let overLap = data.filter((ele,index)=>{
        return data.indexOf(ele) !== index
        })
    overLap = Array.from(new Set(overLap));
    
    return data.filter(function(val) {
        return overLap.indexOf(val) !== -1;
    })
}
