while True:
    print("----Menu----")
    print("1. 올해 윤년 여부")
    print("2. 입력 받은 연도 윤년 여부")
    print("3. 프로그램 종료")
    menu = input("메뉴를 선택하세요 : ")


    if menu == "1":
        current_time = datetime.datetime.now()
        year = current_time.year
        if (year % 4 == 0 and year % 100 !=0) or year % 400 == 0:
            print("윤년입니다.")
        else:
            print("윤년이 아닙니다.")
    elif menu == "2":
        while True:
            try:
                year = input("판별하고자 하는 연도를 입력하세요 : ")
                year = int(year)
                break
            except:
                print("다시 입력하세요.")
        if (year % 4 == 0 and year % 100 !=0) or year % 400 == 0:
            print("윤년입니다.")
        else:
            print("윤년이 아닙니다.")
    elif menu == "3":
        break
        # exit()
    else:
        print("없는 메뉴 입니다.")
